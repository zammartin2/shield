2. src/index.ts
typescript
import express from 'express'
import { createShield } from './config/shield'
import apiRoutes from './routes/api'
import adminRoutes from './routes/admin'
import publicRoutes from './routes/public'
import { authMiddleware } from './middleware/auth'
import { loggingMiddleware } from './middleware/logging'
import dotenv from 'dotenv'

// Загружаем переменные окружения
dotenv.config()

const app = express()
const port = process.env.PORT || 3000

// ============================================
// MIDDLEWARE
// ============================================

// 1. Логирование
app.use(loggingMiddleware)

// 2. Body parsing
app.use(express.json({ limit: '10mb' }))
app.use(express.urlencoded({ extended: true, limit: '10mb' }))

// 3. FAB Shield (полная защита)
const shield = createShield()
app.use(shield.middleware())

// 4. Аутентификация
app.use(authMiddleware)

// ============================================
// МАРШРУТЫ
// ============================================

// Публичные
app.use('/', publicRoutes)

// API (с защитой)
app.use('/api', apiRoutes)

// Админка (с дополнительной защитой)
app.use('/admin', adminRoutes)

// Health check
app.get('/health', (req, res) => {
    res.json({
        status: 'ok',
        service: 'advanced-example',
        version: '1.0.0',
        shield: {
            version: shield.getVersion(),
            active: shield.isActive(),
            metrics: shield.getMetrics()
        },
        timestamp: new Date().toISOString()
    })
})

// Метрики для Prometheus
app.get('/metrics', async (req, res) => {
    const metrics = shield.exportMetrics('prometheus')
    res.set('Content-Type', 'text/plain')
    res.send(metrics)
})

// CSP отчеты
app.post('/api/csp-report', (req, res) => {
    console.log('CSP Violation:', req.body)
    res.status(204).end()
})

// ============================================
// ОБРАБОТКА ОШИБОК
// ============================================

app.use((err: any, req: any, res: any, next: any) => {
    console.error('❌ Error:', err)
    
    const status = err.status || 500
    const message = err.message || 'Internal server error'
    
    res.status(status).json({
        error: message,
        status: status,
        timestamp: new Date().toISOString(),
        requestId: req.id
    })
})

// 404
app.use((req, res) => {
    res.status(404).json({
        error: 'Not found',
        path: req.path,
        timestamp: new Date().toISOString()
    })
})

// ============================================
// ЗАПУСК
// ============================================

app.listen(port, () => {
    console.log('\n🚀 Advanced Example started successfully!')
    console.log(`📍 URL: http://localhost:${port}`)
    console.log(`🛡️ FAB Shield: ${shield.isActive() ? '✅ Active' : '❌ Inactive'}`)
    console.log(`📊 Metrics: http://localhost:${port}/metrics`)
    console.log(`💚 Health: http://localhost:${port}/health`)
    console.log(`🌍 Environment: ${process.env.NODE_ENV || 'development'}`)
    
    const metrics = shield.getMetrics()
    console.log('\n📊 Initial Metrics:')
    console.log(`  Headers: ${metrics.headers?.length || 0}+`)
    console.log(`  Threats blocked: ${metrics.threatsBlocked || 0}`)
    console.log(`  AI models: ${metrics.aiModels || 0}`)
    console.log(`  Plugins: ${metrics.plugins || 0}`)
})
