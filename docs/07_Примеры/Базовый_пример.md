## 🎯 Минимальный пример

### 1. `app.ts`

```typescript
import express from 'express'
import { FABShield } from '@fab-registry/shield'

const app = express()
const port = 3000

// 1. Создаем экземпляр FAB Shield
const shield = new FABShield()

// 2. Подключаем защиту
app.use(shield.middleware())

// 3. Ваш код
app.get('/', (req, res) => {
    res.json({ message: 'Hello from FAB Shield!' })
})

app.listen(port, () => {
    console.log(`🚀 Server running at http://localhost:${port}`)
    console.log(`🛡️ FAB Shield active!`)
})
```

## 🎯 Расширенный пример

```typescript
import express from 'express'
import { FABShield } from '@fab-registry/shield'

const app = express()
const port = 3000

// Создаем экземпляр с минимальной конфигурацией
const shield = new FABShield({
    // Включаем AI-защиту
    ai: {
        enabled: true,
        anomalyDetection: true,
        threatPrediction: true
    },

    // Включаем мониторинг
    monitoring: {
        enabled: true,
        export: ['json']
    },

    // Rate Limiting
    rateLimit: {
        enabled: true,
        default: {
            windowMs: 60000,
            max: 100
        }
    }
})

// Подключаем защиту
app.use(shield.middleware())

// Health check
app.get('/health', (req, res) => {
    res.json({
        status: 'ok',
        shield: {
            version: shield.getVersion(),
            active: shield.isActive()
        }
    })
})

// Метрики
app.get('/metrics', (req, res) => {
    const metrics = shield.getMetrics()
    res.json(metrics)
})

// Ваши маршруты
app.get('/', (req, res) => {
    res.json({
        message: 'Hello from FAB Shield!',
        protected: true,
        features: [
            '✅ Security Headers',
            '✅ CSP',
            '✅ AI Protection',
            '✅ Rate Limiting',
            '✅ Monitoring'
        ]
    })
})

// Обработка ошибок
app.use((err: any, req: any, res: any, next: any) => {
    console.error('Error:', err)

    res.status(500).json({
        error: 'Internal server error'
    })
})

app.listen(port, () => {
    console.log('\n🚀 Server started!')
    console.log(`📍 http://localhost:${port}`)
    console.log(`🛡️ FAB Shield: ${shield.isActive() ? '✅ Active' : '❌ Inactive'}`)

    console.log('\n📋 Endpoints:')
    console.log(`  GET  /`)
    console.log(`  GET  /health`)
    console.log(`  GET  /metrics`)
})
```
