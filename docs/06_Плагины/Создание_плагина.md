## 📋 Введение

**Создание плагина** — это лучший способ расширить функциональность FAB Shield. В этом руководстве мы шаг за шагом создадим полноценный плагин.

---

## 🎯 Что такое плагин?

**Плагин** — это независимый модуль, который:
- 🔌 Расширяет функциональность FAB Shield
- 🎯 Решает конкретную задачу
- 📦 Может использоваться в разных проектах
- 🔄 Обновляется независимо от ядра

Шаг 3: Плагин
typescript
// index.ts

import { Plugin, PluginContext } from '@fab-registry/shield'

// Конфигурация плагина
export interface MyPluginConfig {
    enabled: boolean
    logLevel: 'debug' | 'info' | 'warn' | 'error'
    customOption: string
    // Добавьте свои опции
}

// Конфигурация по умолчанию
const defaultConfig: MyPluginConfig = {
    enabled: true,
    logLevel: 'info',
    customOption: 'default'
}

// Создаем плагин
export const myPlugin: Plugin = {
    name: 'my-plugin',
    version: '1.0.0',
    description: 'Мой первый плагин для FAB Shield',
    author: 'DEVORBIT LLC',
    license: 'MIT',
    
    config: defaultConfig,
    
    // Инициализация
    onInit: async (context: PluginContext) => {
        const config = context.getConfig('my-plugin') as MyPluginConfig
        
        // Проверяем конфигурацию
        if (!config.customOption) {
            throw new Error('customOption is required')
        }
        
        // Логируем
        context.log('info', '✅ Плагин инициализирован', { config })
        
        // Сохраняем в контексте
        context.set('initialized', true)
        context.set('startTime', Date.now())
    },
    
    // Запуск
    onStart: async (context: PluginContext) => {
        const config = context.getConfig('my-plugin') as MyPluginConfig
        
        if (!config.enabled) {
            context.log('warn', '⚠️ Плагин отключен в конфигурации')
            return
        }
        
        context.log('info', '🚀 Плагин запущен')
    },
    
    // Middleware
    middleware: (req, res, next) => {
        const config = context.getConfig('my-plugin') as MyPluginConfig
        
        // Логируем запрос
        context.log('debug', `📝 ${req.method} ${req.path}`)
        
        // Ваша логика здесь
        // Например, добавляем заголовок
        res.setHeader('X-My-Plugin', 'active')
        
        next()
    },
    
    // Обработчик запроса
    onRequest: async (req, context) => {
        const config = context.getConfig('my-plugin') as MyPluginConfig
        
        // Анализируем запрос
        // Возвращаем результат
        return {
            block: false,
            data: {
                processed: true,
                timestamp: new Date()
            }
        }
    },
    
    // Обработчик ответа
    onResponse: async (res, context) => {
        const config = context.getConfig('my-plugin') as MyPluginConfig
        
        // Логируем ответ
        context.log('debug', `📤 ${res.statusCode}`)
    },
    
    // Обработчик ошибок
    onError: async (error, context) => {
        const config = context.getConfig('my-plugin') as MyPluginConfig
        
        context.log('error', '❌ Ошибка в плагине', {
            error: error.message,
            stack: error.stack
        })
    },
    
    // Остановка
    onStop: async (context) => {
        const config = context.getConfig('my-plugin') as MyPluginConfig
        const startTime = context.get('startTime')
        const duration = Date.now() - startTime
        
        context.log('info', `🛑 Плагин остановлен`, {
            duration: `${duration}ms`,
            requests: context.get('requestCount') || 0
        })
    },
    
    // Уничтожение
    onDestroy: async (context) => {
        // Освобождаем ресурсы
        context.log('info', '💀 Плагин уничтожен')
    },
    
    // API методы
    api: {
        getStatus: async (context) => {
            const config = context.getConfig('my-plugin') as MyPluginConfig
            return {
                name: 'my-plugin',
                version: '1.0.0',
                enabled: config.enabled,
                uptime: Date.now() - context.get('startTime')
            }
        },
        getStats: async (context) => {
            return {
                requests: context.get('requestCount') || 0,
                errors: context.get('errorCount') || 0
            }
        }
    }
}

export default myPlugin

// tests/plugin.test.ts

import { myPlugin } from '../index'

describe('My Plugin', () => {
    let context: any
    
    beforeEach(() => {
        context = {
            getConfig: jest.fn().mockReturnValue({
                enabled: true,
                logLevel: 'info',
                customOption: 'test'
            }),
            log: jest.fn(),
            set: jest.fn(),
            get: jest.fn(),
            getShield: jest.fn()
        }
    })
    
    test('should initialize correctly', async () => {
        await myPlugin.onInit(context)
        expect(context.set).toHaveBeenCalledWith('initialized', true)
    })
    
    test('should start correctly', async () => {
        await myPlugin.onStart(context)
        expect(context.log).toHaveBeenCalledWith('info', '🚀 Плагин запущен')
    })
    
    test('should process request', async () => {
        const req = { method: 'GET', path: '/test' }
        const result = await myPlugin.onRequest(req, context)
        expect(result.block).toBe(false)
        expect(result.data.processed).toBe(true)
    })
})
🚀 Использование плагина
Подключение в проекте
typescript
import { FABShield } from '@fab-registry/shield'
import { myPlugin } from '@fab-registry/plugin-my-plugin'

const shield = new FABShield({
    plugins: [
        myPlugin({
            enabled: true,
            logLevel: 'debug',
            customOption: 'my-value'
        })
    ]
})

app.use(shield.middleware())
Обновление конфигурации
typescript
// Обновить конфигурацию плагина
await shield.plugins.updateConfig('my-plugin', {
    logLevel: 'debug',
    customOption: 'new-value'
})

// Получить текущую конфигурацию
const config = shield.plugins.getConfig('my-plugin')
console.log('Текущая конфигурация:', config)
Использование API плагина
typescript
// Вызов API метода плагина
const status = await shield.plugins.callAPI('my-plugin', 'getStatus')
console.log('Статус плагина:', status)

// Получить статистику
const stats = await shield.plugins.callAPI('my-plugin', 'getStats')
console.log('Статистика:', stats)
