### 1. Логирование запросов

```typescript
// plugins/request-logger.ts

export const requestLoggerPlugin = {
    name: 'request-logger',
    version: '1.0.0',
    description: 'Логирование всех запросов',
    author: 'DEVORBIT LLC',
    
    middleware: (req, res, next) => {
        const start = Date.now()
        
        // Логируем запрос
        console.log(`📥 ${req.method} ${req.path} - ${req.ip}`)
        
        // Логируем ответ
        res.on('finish', () => {
            const duration = Date.now() - start
            console.log(`📤 ${req.method} ${req.path} - ${res.statusCode} (${duration}ms)`)
        })
        
        next()
    }
}

2. Добавление кастомных заголовков
typescript
// plugins/custom-headers.ts

export const customHeadersPlugin = {
    name: 'custom-headers',
    version: '1.0.0',
    description: 'Добавление кастомных security-заголовков',
    
    config: {
        headers: {
            'X-Frame-Options': 'DENY',
            'X-Content-Type-Options': 'nosniff',
            'X-XSS-Protection': '1; mode=block',
            'Referrer-Policy': 'strict-origin-when-cross-origin'
        },
        removeHeaders: ['X-Powered-By']
    },
    
    onInit: async (context) => {
        const config = context.getConfig('custom-headers')
        console.log('🔧 Кастомные заголовки загружены:', config.headers)
    },
    
    middleware: (req, res, next) => {
        const config = context.getConfig('custom-headers')
        
        // Добавляем заголовки
        Object.entries(config.headers).forEach(([name, value]) => {
            res.setHeader(name, value)
        })
        
        // Удаляем заголовки
        config.removeHeaders.forEach(name => {
            res.removeHeader(name)
        })
        
        next()
    }
}
3. IP-фильтрация
typescript
// plugins/ip-filter.ts

export const ipFilterPlugin = {
    name: 'ip-filter',
    version: '1.0.0',
    description: 'Фильтрация IP-адресов',
    
    config: {
        whitelist: ['10.0.0.1', '10.0.0.2', '192.168.1.100'],
        blacklist: ['1.2.3.4', '5.6.7.8'],
        mode: 'whitelist'  // 'whitelist' | 'blacklist'
    },
    
    middleware: (req, res, next) => {
        const config = context.getConfig('ip-filter')
        const ip = req.ip
        
        const isWhitelisted = config.whitelist.includes(ip)
        const isBlacklisted = config.blacklist.includes(ip)
        
        if (config.mode === 'whitelist' && !isWhitelisted) {
            return res.status(403).json({
                error: 'IP not whitelisted',
                ip: ip
            })
        }
        
        if (config.mode === 'blacklist' && isBlacklisted) {
            return res.status(403).json({
                error: 'IP is blacklisted',
                ip: ip
            })
        }
        
        next()
    }
}

4. Мониторинг производительности
typescript
// plugins/performance-monitor.ts

import { performance } from 'perf_hooks'

export const performancePlugin = {
    name: 'performance-monitor',
    version: '1.0.0',
    description: 'Мониторинг производительности',
    
    config: {
        thresholds: {
            warning: 500,   // ms
            error: 1000     // ms
        },
        enabled: true
    },
    
    onInit: async (context) => {
        const config = context.getConfig('performance-monitor')
        console.log('📊 Мониторинг производительности включен')
    },
    
    middleware: (req, res, next) => {
        const start = performance.now()
        
        res.on('finish', () => {
            const duration = performance.now() - start
            const config = context.getConfig('performance-monitor')
            
            if (duration > config.thresholds.error) {
                console.error(`❌ Медленный запрос: ${req.path} - ${duration.toFixed(2)}ms`)
                context.emit('performance:error', {
                    path: req.path,
                    method: req.method,
                    duration: duration,
                    threshold: config.thresholds.error
                })
            } else if (duration > config.thresholds.warning) {
                console.warn(`⚠️ Медленный запрос: ${req.path} - ${duration.toFixed(2)}ms`)
                context.emit('performance:warning', {
                    path: req.path,
                    method: req.method,
                    duration: duration,
                    threshold: config.thresholds.warning
                })
            }
        })
        
        next()
    }
}
5. Rate Limiting с уведомлениями
typescript
// plugins/smart-rate-limit.ts

import { RateLimiter } from 'rate-limiter-flexible'

export const smartRateLimitPlugin = {
    name: 'smart-rate-limit',
    version: '1.0.0',
    description: 'Умное ограничение запросов с уведомлениями',
    
    config: {
        defaultLimit: 100,
        window: 60,  // секунд
        userLimits: {
            admin: 1000,
            user: 100,
            guest: 50
        },
        notifications: {
            email: 'admin@company.com',
            slack: process.env.SLACK_WEBHOOK_URL
        }
    },
    
    onInit: async (context) => {
        const config = context.getConfig('smart-rate-limit')
        
        // Инициализируем rate limiter
        const limiter = new RateLimiter({
            points: config.defaultLimit,
            duration: config.window,
            keyPrefix: 'ratelimit'
        })
        
        context.limiter = limiter
        context.limits = config.userLimits
    },
    
    onRequest: async (req, context) => {
        const key = req.user?.id || req.ip
        const limit = context.limits[req.user?.role || 'guest']
        
        try {
            const result = await context.limiter.consume(key)
            
            // Добавляем заголовки
            res.setHeader('X-RateLimit-Limit', limit)
            res.setHeader('X-RateLimit-Remaining', result.remainingPoints)
            res.setHeader('X-RateLimit-Reset', new Date(Date.now() + result.msBeforeNext).toISOString())
            
            return { block: false }
        } catch (error) {
            // Превышен лимит
            const config = context.getConfig('smart-rate-limit')
            
            // Отправляем уведомление
            await sendNotification('Rate limit exceeded', {
                ip: req.ip,
                path: req.path,
                limit: limit
            })
            
            return {
                block: true,
                status: 429,
                message: 'Too many requests',
                headers: {
                    'Retry-After': Math.ceil(error.msBeforeNext / 1000)
                }
            }
        }
    }
}

// plugins/telegram-notifier.ts

import { Telegraf } from 'telegraf'

export const telegramPlugin = {
    name: 'telegram-notifier',
    version: '1.0.0',
    description: 'Уведомления в Telegram',
    
    config: {
        botToken: process.env.TELEGRAM_BOT_TOKEN,
        chatId: process.env.TELEGRAM_CHAT_ID,
        events: ['threat', 'error', 'warning']
    },
    
    onInit: async (context) => {
        const config = context.getConfig('telegram-notifier')
        const bot = new Telegraf(config.botToken)
        
        context.bot = bot
        context.chatId = config.chatId
        
        // Проверяем подключение
        try {
            await bot.telegram.getMe()
            console.log('✅ Telegram бот подключен')
        } catch (error) {
            console.error('❌ Ошибка подключения Telegram бота:', error)
        }
    },
    
    onError: async (error, context) => {
        const config = context.getConfig('telegram-notifier')
        
        await context.bot.telegram.sendMessage(
            context.chatId,
            `🚨 Ошибка в FAB Shield:\n\n` +
            `Сообщение: ${error.message}\n` +
            `Время: ${new Date().toISOString()}`
        )
    },
    
    onRequest: async (req, context) => {
        // Проверяем угрозы
        const shield = context.getShield()
        const metrics = shield.getMetrics()
        
        if (metrics.threatsBlocked > 10) {
            await context.bot.telegram.sendMessage(
                context.chatId,
                `⚠️ Обнаружено ${metrics.threatsBlocked} угроз!\n` +
                `IP: ${req.ip}\n` +
                `Путь: ${req.path}`
            )
        }
    }
}
7. Аудит с сохранением в базу
typescript
// plugins/audit-database.ts

import { PrismaClient } from '@prisma/client'

export const auditDatabasePlugin = {
    name: 'audit-database',
    version: '1.0.0',
    description: 'Сохранение аудит-логов в базу данных',
    
    config: {
        databaseUrl: process.env.DATABASE_URL,
        tableName: 'audit_logs',
        batchSize: 100,
        flushInterval: 60  // секунд
    },
    
    onInit: async (context) => {
        const config = context.getConfig('audit-database')
        
        // Инициализируем Prisma
        const prisma = new PrismaClient({
            datasources: {
                db: {
                    url: config.databaseUrl
                }
            }
        })
        
        context.prisma = prisma
        context.queue = []
        
        // Запускаем периодическую очистку
        setInterval(() => {
            context.flushQueue()
        }, config.flushInterval * 1000)
    },
    
    onRequest: async (req, context) => {
        const log = {
            type: 'request',
            method: req.method,
            path: req.path,
            ip: req.ip,
            userAgent: req.headers['user-agent'],
            userId: req.user?.id,
            timestamp: new Date()
        }
        
        context.queue.push(log)
        
        // Если очередь заполнена - сохраняем
        if (context.queue.length >= context.config.batchSize) {
            await context.flushQueue()
        }
    },
    
    onError: async (error, context) => {
        const log = {
            type: 'error',
            message: error.message,
            stack: error.stack,
            timestamp: new Date()
        }
        
        context.queue.push(log)
        await context.flushQueue()
    },
    
    flushQueue: async function() {
        if (this.queue.length === 0) return
        
        try {
            await this.prisma.auditLog.createMany({
                data: this.queue
            })
            this.queue = []
        } catch (error) {
            console.error('❌ Ошибка сохранения аудит-логов:', error)
        }
    }
}
8. Кастомные правила безопасности
typescript
// plugins/custom-rules.ts

export const customRulesPlugin = {
    name: 'custom-rules',
    version: '1.0.0',
    description: 'Кастомные правила безопасности',
    
    config: {
        rules: [
            {
                name: 'block-common-attacks',
                patterns: [
                    /(SELECT|INSERT|UPDATE|DELETE|DROP|UNION)/i,
                    /<script>.*<\/script>/i,
                    /\.\.\/\.\.\//,
                    /;.*rm -rf/,
                    /wget.*http/
                ],
                action: 'block',
                severity: 'high'
            },
            {
                name: 'admin-access',
                path: '/admin/*',
                requireAuth: true,
                requireRole: 'admin',
                action: 'block'
            },
            {
                name: 'api-rate-limit',
                path: '/api/*',
                rateLimit: {
                    window: 60,
                    max: 100
                },
                action: 'block'
            }
        ]
    },
    
    onInit: async (context) => {
        const config = context.getConfig('custom-rules')
        console.log(`🔧 Загружено правил: ${config.rules.length}`)
    },
    
    middleware: (req, res, next) => {
        const config = context.getConfig('custom-rules')
        
        for (const rule of config.rules) {
            // Проверяем path
            if (rule.path && !req.path.match(rule.path)) continue
            
            // Проверяем паттерны
            if (rule.patterns) {
                const body = req.body ? JSON.stringify(req.body) : ''
                const query = req.url?.split('?')[1] || ''
                const combined = body + query
                
                for (const pattern of rule.patterns) {
                    if (pattern.test(combined)) {
                        res.status(403).json({
                            error: 'Blocked by custom rule',
                            rule: rule.name,
                            pattern: pattern.toString()
                        })
                        return
                    }
                }
            }
            
            // Проверяем авторизацию
            if (rule.requireAuth && !req.user) {
                res.status(401).json({
                    error: 'Authentication required',
                    rule: rule.name
                })
                return
            }
            
            // Проверяем роль
            if (rule.requireRole && req.user?.role !== rule.requireRole) {
                res.status(403).json({
                    error: 'Insufficient permissions',
                    rule: rule.name,
                    requiredRole: rule.requireRole
                })
                return
            }
        }
        
        next()
    }
}
9. Интеграция с внешним API
typescript
// plugins/external-api.ts

import axios from 'axios'

export const externalApiPlugin = {
    name: 'external-api',
    version: '1.0.0',
    description: 'Интеграция с внешним API безопасности',
    
    config: {
        apiUrl: 'https://api.security.com',
        apiKey: process.env.SECURITY_API_KEY,
        timeout: 5000,
        checkIP: true,
        checkUserAgent: true
    },
    
    onInit: async (context) => {
        const config = context.getConfig('external-api')
        
        // Инициализируем клиент
        const client = axios.create({
            baseURL: config.apiUrl,
            headers: {
                'Authorization': `Bearer ${config.apiKey}`
            },
            timeout: config.timeout
        })
        
        context.client = client
    },
    
    onRequest: async (req, context) => {
        const config = context.getConfig('external-api')
        
        try {
            // Проверяем IP
            if (config.checkIP) {
                const ipResponse = await context.client.get(`/ip/${req.ip}`)
                if (ipResponse.data.blocked) {
                    return {
                        block: true,
                        status: 403,
                        message: 'IP blocked by external service',
                        reason: ipResponse.data.reason
                    }
                }
            }
            
            // Проверяем User-Agent
            if (config.checkUserAgent) {
                const uaResponse = await context.client.post('/user-agent', {
                    userAgent: req.headers['user-agent']
                })
                
                if (uaResponse.data.suspicious) {
                    return {
                        block: true,
                        status: 403,
                        message: 'Suspicious user-agent',
                        reason: uaResponse.data.reason
                    }
                }
            }
        } catch (error) {
            // В случае ошибки внешнего API — пропускаем
            console.warn('⚠️ External API error:', error.message)
        }
    }
}
