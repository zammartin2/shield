## 🚀 Базовые примеры

### 1. Минимальная конфигурация

Самый простой способ начать использовать FAB Shield:

```typescript
import express from 'express'
import { FABShield } from '@fab-registry/shield'

const app = express()
const shield = new FABShield()

app.use(shield.middleware())

app.get('/', (req, res) => {
    res.json({ message: 'Hello World!' })
})

app.listen(3000)
2. С AI-защитой
Включаем интеллектуальную защиту от угроз:

🎯 Примеры для конкретных задач
1. Защита API
typescript
import express from 'express'
import { FABShield } from '@fab-registry/shield'

const app = express()

const shield = new FABShield({
    // Строгий CSP для API
    csp: {
        enabled: true,
        directives: {
            'default-src': ["'none'"],
            'script-src': ["'none'"],
            'style-src': ["'none'"],
            'img-src': ["'none'"]
        }
    },
    
    // AI-защита
    ai: {
        enabled: true,
        sqlInjectionProtection: true,
        xssProtection: true
    },
    
    // Rate limiting
    rules: [
        {
            path: '/api/*',
            action: 'rate-limit',
            config: {
                maxRequests: 100,
                windowMs: 60000
            }
        }
    ]
})

app.use(shield.middleware())

// Защищенное API
app.get('/api/data', (req, res) => {
    res.json({ data: 'secret' })
})

app.listen(3000)
2. Защита админ-панели
typescript
import express from 'express'
import { FABShield } from '@fab-registry/shield'

const app = express()

const shield = new FABShield({
    // Максимальная защита для админки
    csp: {
        enabled: true,
        strict: true,
        directives: {
            'default-src': ["'self'"],
            'script-src': ["'self'", "'strict-dynamic'"],
            'style-src': ["'self'"],
            'img-src': ["'self'", "data:"]
        }
    },
    
    // Дополнительные заголовки
    headers: {
        hsts: {
            maxAge: 31536000,
            includeSubDomains: true,
            preload: true
        }
    },
    
    // Правила для админки
    rules: [
        {
            path: '/admin/*',
            action: 'challenge',
            condition: (req) => !req.session?.isAdmin
        }
    ]
})

app.use(shield.middleware())

app.get('/admin', (req, res) => {
    res.send('Admin Panel')
})

app.listen(3000)
3. Защита в production
typescript
import express from 'express'
import { FABShield } from '@fab-registry/shield'

const app = express()

const shield = new FABShield({
    // Production настройки
    env: 'production',
    
    // Максимальная безопасность
    csp: {
        enabled: true,
        dynamic: true,
        strict: true,
        reportOnly: false
    },
    
    headers: {
        hsts: {
            maxAge: 31536000,
            includeSubDomains: true,
            preload: true
        }
    },
    
    ai: {
        enabled: true,
        anomalyDetection: true,
        threatPrediction: true,
        autoAdapt: true
    },
    
    monitoring: {
        enabled: true,
        export: ['prometheus', 'json'],
        alerts: {
            enabled: true,
            rules: [
                {
                    metric: 'threats_blocked',
                    threshold: 10,
                    interval: 60,
                    severity: 'high'
                }
            ]
        }
    },
    
    logging: {
        level: 'info',
        format: 'json',
        transports: [
            { type: 'console', enabled: true },
            { type: 'file', enabled: true, path: './logs/shield.log' }
        ]
    }
})

app.use(shield.middleware())

app.get('/', (req, res) => {
    res.json({ message: 'Production ready!' })
})

app.listen(3000)
4. С кастомными правилами
typescript
import express from 'express'
import { FABShield } from '@fab-registry/shield'

const app = express()

const shield = new FABShield({
    // Кастомные правила безопасности
    rules: [
        // Блокировка по IP
        {
            path: '*',
            action: 'block',
            condition: (req) => {
                const blockedIPs = ['192.168.1.100']
                return blockedIPs.includes(req.ip)
            }
        },
        
        // Защита от ботов
        {
            path: '/api/*',
            action: 'challenge',
            condition: (req) => {
                const userAgent = req.headers['user-agent'] || ''
                const bots = ['curl', 'wget', 'python-requests']
                return bots.some(b => userAgent.includes(b))
            }
        },
        
        // Ограничение по времени
        {
            path: '/api/*',
            action: 'block',
            condition: (req) => {
                const hour = new Date().getHours()
                return hour < 6 || hour > 23 // Блокировка ночью
            }
        }
    ]
})

app.use(shield.middleware())

app.get('/', (req, res) => {
    res.json({ message: 'With custom rules!' })
})

app.listen(3000)
🔌 Примеры с плагинами
1. Кастомный плагин
typescript
import express from 'express'
import { FABShield } from '@fab-registry/shield'

const app = express()

// Создаем плагин
const customPlugin = {
    name: 'custom-security',
    version: '1.0.0',
    middleware: (req, res, next) => {
        // Добавляем кастомный заголовок

const shield = new FABShield({
    plugins: [customPlugin]
})

app.use(shield.middleware())

app.get('/', (req, res) => {

})

app.listen(3000)
2. Плагин для аудита
typescript
import express from 'express'
import { FABShield } from '@fab-registry/shield'
import fs from 'fs'

const app = express()

const auditPlugin = {
    name: 'audit-logger',
    middleware: (req, res, next) => {
        const log = {
            timestamp: new Date().toISOString(),
            method: req.method,
            url: req.url,
            ip: req.ip,
            userAgent: req.headers['user-agent']
        }
        
        fs.appendFileSync(
            './audit.log',
            JSON.stringify(log) + '\n'
        )
        
        next()
    }
}

const shield = new FABShield({
    plugins: [auditPlugin]
})

app.use(shield.middleware())

app.get('/', (req, res) => {
    res.json({ message: 'Audit logged!' })
})

app.listen(3000)
3. Плагин для Telegram уведомлений
typescript
import express from 'express'
import { FABShield } from '@fab-registry/shield'
import axios from 'axios'

const app = express()

const telegramPlugin = {
    name: 'telegram-alerts',
    middleware: async (req, res, next) => {
        // Проверяем, есть ли угрозы
        const shield = (req as any).shield
        if (shield && shield.threatsDetected > 0) {
            await axios.post(
                `https://api.telegram.org/bot${process.env.TELEGRAM_BOT_TOKEN}/sendMessage`,
                {
                    chat_id: process.env.TELEGRAM_CHAT_ID,
                    text: `🚨 Threat detected!\nIP: ${req.ip}\nPath: ${req.url}`
                }
            )
        }
        next()
    }
}

const shield = new FABShield({
    plugins: [telegramPlugin]
})

app.use(shield.middleware())

app.listen(3000)

📊 Пример с метриками и Grafana
typescript
import express from 'express'
import { FABShield } from '@fab-registry/shield'
import prometheus from 'prom-client'

const app = express()

const shield = new FABShield({
    monitoring: {
        enabled: true,
        export: ['prometheus']
    }
})

app.get('/metrics', async (req, res) => {

})
