## 🚀 Базовые примеры

### 1. Минимальная конфигурация

Самый простой способ начать использовать FAB Shield:

```typescript
import express from 'express'
import { FABShield } from '@fab-orbita/shield'

const app = express()
const shield = new FABShield()

app.use(shield.middleware())

app.get('/', (req, res) => {
  res.json({
    message: 'Hello World!'
  })
})

app.listen(3000, () => {
  console.log('🚀 Server started on http://localhost:3000')
})
```

---

### 2. С AI-защитой

Включаем интеллектуальную защиту от подозрительных запросов:

```typescript
import express from 'express'
import { FABShield } from '@fab-orbita/shield'

const app = express()

const shield = new FABShield({

    threatPrediction: true

  res.json({
    message: 'Protected by AI!'
  })
})

app.listen(3000, () => {
  console.log('🤖 AI protection enabled')
})
```

### 3. С мониторингом

Добавляем сбор метрик и endpoint для просмотра статистики:

## Express.js

```typescript
import express from 'express'
import { FABShield } from '@fab-orbita/shield'

const app = express()

app.use(express.json())

const shield = new FABShield({
  ai: {
    enabled: true
  },

  monitoring: {
    enabled: true
  }
})

// Подключаем защиту до маршрутов
app.use(shield.middleware())

app.get('/api/users', (req, res) => {
  res.json({
    users: []
  })
})

app.post('/api/users', (req, res) => {
  res.status(201).json({
    id: 1
  })
})

app.listen(3000, () => {
  console.log('🚀 Express app protected by FAB Shield')
})
```

## Fastify

> Если используется Express-style middleware, для Fastify может потребоваться `@fastify/middie`.

Установка:

```bash
npm install fastify @fastify/middie @fab-orbita/shield
```

Пример:

```typescript
import Fastify from 'fastify'
import middie from '@fastify/middie'
import { FABShield } from '@fab-orbita/shield'

const app = Fastify()

const shield = new FABShield({
  ai: {
    enabled: true
  }
})

await app.register(middie)

// Подключаем middleware
app.use(shield.middleware())

app.get('/', async () => {
  return {
    message: 'Hello World!'
  }
})

await app.listen({
  port: 3000
})

console.log('🚀 Fastify app protected by FAB Shield')
```

## Koa

```typescript
import Koa from 'koa'
import { FABShield } from '@fab-orbita/shield'

const app = new Koa()

const shield = new FABShield({
  ai: {
    enabled: true
  }
})

// Если в вашей версии FAB Shield доступен koaMiddleware:

app.use(async (ctx) => {
  ctx.body = {
    message: 'Hello World!'
  }
})

app.listen(3000, () => {
  console.log('🚀 Koa app protected by FAB Shield')
})
```

> Если используется только Express-style middleware, подключите адаптер middleware для Koa или используйте соответствующий adapter из FAB Shield.

## 🎯 Примеры для конкретных задач

### 1. Защита API

```typescript
import express from 'express'
import { FABShield } from '@fab-orbita/shield'

const app = express()

app.use(express.json())

const shield = new FABShield({
  // Строгий CSP для API
  csp: {
    enabled: true,
    directives: {
      'default-src': ["'none'"],
      'script-src': ["'none'"],
      'style-src': ["'none'"],
      'img-src': ["'none'"],
      'frame-ancestors': ["'none'"],
      'base-uri': ["'none'"],
      'form-action': ["'none'"]
    }
  },

  // AI-защита
  ai: {
    enabled: true,
    sqlInjectionProtection: true,
    xssProtection: true
  },

  // Rate limiting
  rules: [
    {
      path: '/api/*',
      action: 'rate-limit',
      config: {
        maxRequests: 100,
        windowMs: 60_000
      }
    }
  ]
})

app.use(shield.middleware())

app.get('/api/data', (req, res) => {
  res.json({
    data: 'secret'
  })
})

app.listen(3000, () => {
  console.log('🛡️ API protected')
})
```

### 2. Защита админ-панели

```typescript
import express from 'express'
import session from 'express-session'
import { FABShield } from '@fab-orbita/shield'

const app = express()

app.use(session({
  secret: process.env.SESSION_SECRET || 'change-me',
  resave: false,
  saveUninitialized: false,
  cookie: {
    httpOnly: true,
    sameSite: 'lax',
    secure: process.env.NODE_ENV === 'production'
  }
}))

const shield = new FABShield({
  csp: {
    enabled: true,
    strict: true,
    directives: {
      'default-src': ["'self'"],
      'script-src': ["'self'"],
      'style-src': ["'self'"],
      'img-src': ["'self'", 'data:'],
      'frame-ancestors': ["'none'"]
    }
  },

  headers: {
    hsts: {
      maxAge: 31536000,
      includeSubDomains: true,
      preload: true
    }
  },

  rules: [
    {
      path: '/admin/*',
      action: 'challenge',
      condition: (req) => !req.session?.isAdmin
    }
  ]
})

app.use(shield.middleware())

app.get('/admin', (req, res) => {
  res.send('Admin Panel')
})

app.listen(3000, () => {
  console.log('🔐 Admin panel protected')
})
```

### 3. Защита в production

```typescript
import express from 'express'
import { FABShield } from '@fab-orbita/shield'

const app = express()

app.use(express.json())

const shield = new FABShield({
  env: 'production',

  csp: {
    enabled: true,
    dynamic: true,
    strict: true,
    reportOnly: false
  },

  headers: {
    hsts: {
      maxAge: 31536000,
      includeSubDomains: true,
      preload: true
    }
  },

  ai: {
    enabled: true,
    anomalyDetection: true,
    threatPrediction: true,
    autoAdapt: true
  },

  monitoring: {
    enabled: true,
    export: ['prometheus', 'json'],
    alerts: {
      enabled: true,
      rules: [
        {
          metric: 'threats_blocked',
          threshold: 10,
          interval: 60,
          severity: 'high'
        }
      ]
    }
  },

  logging: {
    level: 'info',
    format: 'json',
    transports: [
      {
        type: 'console',
        enabled: true
      },
      {
        type: 'file',
        enabled: true,
        path: './logs/shield.log'
      }
    ]
  }
})

app.use(shield.middleware())

app.get('/', (req, res) => {
  res.json({
    message: 'Production ready!'
  })
})

app.listen(3000, () => {
  console.log('🚀 Production app protected by FAB Shield')
})
```

### 4. Кастомные правила

```typescript
import express from 'express'
import { FABShield } from '@fab-orbita/shield'

const app = express()

const shield = new FABShield({
  rules: [
    // Блокировка по IP
    {
      path: '*',
      action: 'block',
      condition: (req) => {
        const blockedIPs = ['192.168.1.100']
        return blockedIPs.includes(req.ip)
      }
    },

    // Защита от простых скриптовых клиентов
    {
      path: '/api/*',
      action: 'challenge',
      condition: (req) => {
        const userAgent = req.headers['user-agent'] || ''
        const suspiciousClients = ['curl', 'wget', 'python-requests']

        return suspiciousClients.some((client) =>
          userAgent.toLowerCase().includes(client)
        )
      }
    },

    // Ограничение доступа по времени
    {
      path: '/api/private/*',
      action: 'block',
      condition: () => {
        const hour = new Date().getHours()
        return hour < 6 || hour > 23
      }
    }
  ]
})

app.use(shield.middleware())

app.get('/', (req, res) => {
  res.json({
    message: 'With custom rules!'
  })
})

app.listen(3000, () => {
  console.log('⚙️ Custom rules enabled')
})
```

import express from 'express'
import { FABShield } from '@fab-orbita/shield'

### 2. Плагин для аудита

```typescript
import express from 'express'
import fs from 'fs'
import { FABShield } from '@fab-orbita/shield'

const app = express()

const auditPlugin = {
  name: 'audit-logger',

  middleware: (req, res, next) => {
    const log = {
      timestamp: new Date().toISOString(),
      method: req.method,
      url: req.url,
      ip: req.ip,
      userAgent: req.headers['user-agent']
    }

    fs.appendFileSync(
      './audit.log',
      JSON.stringify(log) + '\n'
    )

    next()
  }
}

const shield = new FABShield({
  plugins: [auditPlugin]
})

app.use(shield.middleware())

app.get('/', (req, res) => {
  res.json({
    message: 'Audit logged!'
  })
})

app.listen(3000, () => {
  console.log('🧾 Audit plugin enabled')
})
```

### 3. Плагин для Telegram-уведомлений

```typescript
import express from 'express'
import axios from 'axios'
import { FABShield } from '@fab-orbita/shield'

const app = express()

const telegramPlugin = {
  name: 'telegram-alerts',

  middleware: async (req, res, next) => {
    const shieldContext = (req as any).shield

    if (shieldContext && shieldContext.threatsDetected > 0) {
      await axios.post(
        `https://api.telegram.org/bot${process.env.TELEGRAM_BOT_TOKEN}/sendMessage`,
        {
          chat_id: process.env.TELEGRAM_CHAT_ID,
          text: `🚨 Threat detected!\nIP: ${req.ip}\nPath: ${req.url}`
        }
      )
    }

    next()
  }
}

const shield = new FABShield({
  plugins: [telegramPlugin]
})

app.use(shield.middleware())

app.listen(3000, () => {
  console.log('📲 Telegram alerts enabled')
})
```

> Не храните токены Telegram в коде. Используйте environment variables.

## 📊 Пример с Prometheus и Grafana

```typescript
import express from 'express'
import prometheus from 'prom-client'
import { FABShield } from '@fab-orbita/shield'

const app = express()

const shield = new FABShield({
  monitoring: {
    enabled: true,
    export: ['prometheus']
  }
})

app.use(shield.middleware())

// Endpoint для Prometheus
app.get('/metrics', async (req, res) => {
  const metrics = await prometheus.register.metrics()

  res.set('Content-Type', prometheus.register.contentType)
  res.send(metrics)
})

app.get('/', (req, res) => {
  res.json({
    message: 'Metrics enabled'
  })
})

app.listen(3000, () => {
  console.log('📊 Metrics available on /metrics')
})
```

## ✅ Production-чеклист

Перед запуском в production проверьте:

- [ ] FAB Shield подключен до маршрутов
- [ ] Используется HTTPS
- [ ] Включен HSTS
- [ ] CSP протестирован в `reportOnly`
- [ ] Секреты вынесены в environment variables
- [ ] Логи не содержат токены и пароли
- [ ] Настроен rate limiting
- [ ] Настроен мониторинг
- [ ] Настроены alerts
- [ ] Зависимости обновлены
- [ ] Приложение протестировано на staging
- [ ] Проверены headers через `curl -I`
