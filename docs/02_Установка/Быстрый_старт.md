📝 Полный пример
typescript
// Полный пример с настройками
import express from 'express'
import { FABShield } from '@fab-registry/shield'

const app = express()

// Создаем защиту с настройками
const shield = new FABShield({
    // AI-защита
    ai: {
        enabled: true,
        anomalyDetection: true,
        threatPrediction: true,
        xssProtection: true,
        sqlInjectionProtection: true
    },
    
    // Мониторинг
    monitoring: {
        enabled: true,
        export: ['prometheus', 'json']
    },
    
    // Кастомные правила
    rules: [
        {
            path: '/api/*',
            method: 'POST',
            action: 'block',
            condition: (req) => !req.headers['x-api-key']
        }
    ],
    
    // Доверенные CDN
    trustedCDNs: [
        'https://cdnjs.cloudflare.com',
        'https://cdn.jsdelivr.net'
    ]
})

// Подключаем защиту
app.use(shield.middleware())

// API эндпоинт для метрик
app.get('/metrics', (req, res) => {
    const metrics = shield.getMetrics()
    res.json(metrics)
})

// Основной маршрут
app.get('/', (req, res) => {
    res.json({ 
        message: 'Hello World!',
        protected: true 
    })
})

// Запуск
app.listen(3000, () => {
    console.log('🚀 FAB Shield активен!')
    console.log('🔒 Защита включена на порту 3000')
})

// Graceful shutdown
process.on('SIGTERM', () => {
    console.log('👋 Завершение работы...')
    process.exit(0)
})
