🚀 Способы установки
1. Установка через npm
Основной способ — установка через npm:

bash
npm install @fab-registry/shield
Для production:

bash
npm install @fab-registry/shield --save-prod
Для разработки:

bash
npm install @fab-registry/shield --save-dev
2. Установка через yarn
bash
yarn add @fab-registry/shield
Для production:

bash
yarn add @fab-registry/shield --prod
Для разработки:

bash
yarn add @fab-registry/shield --dev
3. Установка через pnpm
bash
pnpm add @fab-registry/shield
Для production:

bash
pnpm add @fab-registry/shield --prod
Для разработки:

bash
pnpm add @fab-registry/shield --dev
4. Установка через реестр FAB
bash
npm install @fab-registry/shield --registry=https://fab.devorbit.ru
Или настройка реестра:

bash
npm config set @fab-registry:registry https://fab.devorbit.ru
npm install @fab-registry/shield

Проверка через package.json
json

}

javascript

Проверка через CLI
bash
# Проверка версии

⚙️ Настройка окружения
1. Создание .env файла
bash
# .env
SHIELD_AI_ENABLED=true
SHIELD_MONITORING_ENABLED=true
SHIELD_LOG_LEVEL=info
SHIELD_RATE_LIMIT_MAX=100
SHIELD_RATE_LIMIT_WINDOW_MS=60000
2. Настройка TypeScript
json
// tsconfig.json
{
  "compilerOptions": {
    "target": "ES2022",
    "module": "commonjs",
    "strict": true,
    "esModuleInterop": true,
    "skipLibCheck": true,
    "forceConsistentCasingInFileNames": true,
    "resolveJsonModule": true,
    "types": ["node"]
  },
  "include": ["src/**/*"],
  "exclude": ["node_modules", "dist"]
}
3. Настройка ESLint
json
// .eslintrc.json
{
  "extends": [
    "eslint:recommended",
    "plugin:@typescript-eslint/recommended"
  ],
  "parser": "@typescript-eslint/parser",
  "plugins": ["@typescript-eslint"],
  "root": true
}
🐳 Установка в Docker
Dockerfile
dockerfile
FROM node:18-alpine

WORKDIR /app

# Копируем package.json
COPY package*.json ./

# Устанавливаем зависимости
RUN npm install

# Копируем исходники
COPY . .

# Устанавливаем FAB Shield
RUN npm install @fab-registry/shield

EXPOSE 3000

CMD ["node", "app.js"]
Сборка образа
bash
docker build -t my-app .
Запуск контейнера
bash
docker run -p 3000:3000 my-app
🚨 Устранение проблем
Проблема: Ошибка при установке
bash
# Очистить кэш npm
npm cache clean --force

# Удалить node_modules
rm -rf node_modules

# Установить заново
npm install
Проблема: Конфликт версий
bash
# Обновить все зависимости
npm update

# Или установить конкретную версию
npm install @fab-registry/shield@1.0.0
Проблема: TypeScript ошибки
json
// tsconfig.json — добавить
{
  "compilerOptions": {
    "skipLibCheck": true,
    "types": ["node"]
  }
}
Проблема: Реестр недоступен
bash
# Использовать официальный npm
npm install @fab-registry/shield --registry=https://registry.npmjs.org
