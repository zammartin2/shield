## 📦 Типы тестов

### 1. Unit-тесты

**Назначение:** тестирование отдельных компонентов.

**Пример:**

```typescript
// src/__tests__/core/headers.test.ts

import { HeadersModule } from '../../core/headers'

describe('HeadersModule', () => {
    let headers: HeadersModule

    beforeEach(() => {
        headers = new HeadersModule()
    })

    test('should add security headers', () => {
        const req = {} as any
        const res = {
            setHeader: jest.fn(),
            removeHeader: jest.fn()
        } as any

        headers.apply(req, res)

        expect(res.setHeader).toHaveBeenCalledWith(
            'X-Frame-Options',
            'SAMEORIGIN'
        )

        expect(res.setHeader).toHaveBeenCalledWith(
            'X-Content-Type-Options',
            'nosniff'
        )

        expect(res.removeHeader).toHaveBeenCalledWith('X-Powered-By')
    })

    test('should handle custom headers', () => {
        const customHeaders = {
            'X-Custom': 'value'
        }

        headers.setCustomHeaders(customHeaders)

        const req = {} as any
        const res = {
            setHeader: jest.fn()
        } as any

        headers.apply(req, res)

        expect(res.setHeader).toHaveBeenCalledWith('X-Custom', 'value')
    })
})
```

### 2. Интеграционные тесты

**Назначение:** тестирование взаимодействия модулей.

**Пример:**

```typescript
// src/__tests__/integration/shield.test.ts

import { FABShield } from '../../index'
import express from 'express'
import request from 'supertest'

describe('FAB Shield Integration', () => {
    let app: express.Express
    let shield: FABShield

    beforeEach(() => {
        app = express()

        shield = new FABShield({
            headers: { enabled: true },
            csp: { enabled: true }
        })

        app.use(shield.middleware())

        app.get('/test', (req, res) => {
            res.json({ ok: true })
        })
    })

    test('should add security headers to response', async () => {
        const response = await request(app).get('/test')

        expect(response.headers['x-frame-options']).toBe('SAMEORIGIN')
        expect(response.headers['x-content-type-options']).toBe('nosniff')
        expect(response.headers['referrer-policy']).toBeDefined()
    })

    test('should handle rate limiting', async () => {
        shield.updateConfig({
            rateLimit: {
                enabled: true,
                max: 2,
                windowMs: 1000
            }
        })

        const requests = [
            request(app).get('/test'),
            request(app).get('/test'),
            request(app).get('/test')
        ]

        const responses = await Promise.all(requests)

        expect(responses[0].status).toBe(200)
        expect(responses[1].status).toBe(200)
        expect(responses[2].status).toBe(429)
    })
})
```

### 3. E2E-тесты

**Назначение:** тестирование полного сценария работы приложения.

**Пример:**

```typescript
// tests/e2e/security.e2e.test.ts

import { startServer, stopServer } from '../helpers/server'
import axios from 'axios'

describe('E2E Security Tests', () => {
    let baseURL: string

    beforeAll(async () => {
        const server = await startServer()
        baseURL = `http://localhost:${server.port}`
    })

    afterAll(async () => {
        await stopServer()
    })

    test('should block XSS attacks', async () => {
        const response = await axios.post(
            `${baseURL}/api/test`,
            {
                input: '<script>alert(1)</script>'
            },
            {
                validateStatus: () => true
            }
        )

        expect(response.status).toBe(403)
        expect(response.data.error).toContain('XSS')
    })

    test('should block SQL injection', async () => {
        const response = await axios.get(
            `${baseURL}/api/users?search=' OR '1'='1`,
            {
                validateStatus: () => true
            }
        )

        expect(response.status).toBe(403)
        expect(response.data.error).toContain('SQL')
    })

    test('should enforce rate limiting', async () => {
        const requests = []

        for (let i = 0; i < 20; i++) {
            requests.push(
                axios.get(`${baseURL}/api/data`, {
                    validateStatus: () => true
                })
            )
        }

        const responses = await Promise.all(requests)
        const blockedCount = responses.filter(r => r.status === 429).length

        expect(blockedCount).toBeGreaterThan(0)
    })
})
```

// tests/security/dependency-scan.test.ts

import { exec } from 'child_process'
import { promisify } from 'util'

const execAsync = promisify(exec)

describe('Security Tests', () => {
    test('should have no high severity vulnerabilities', async () => {
        const { stdout } = await execAsync('npm audit --json')
        const audit = JSON.parse(stdout)

        const highVulnerabilities = audit.vulnerabilities?.high || 0

        expect(highVulnerabilities).toBe(0)

    test('should have no critical vulnerabilities', async () => {
        const { stdout } = await execAsync('npm audit --json')
        const audit = JSON.parse(stdout)

        expect(criticalVulnerabilities).toBe(0)
    })
})

      - name: Upload coverage
        uses: codecov/codecov-action@v3
