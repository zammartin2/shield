## 📦 Способы сборки

### 1. Сборка через npm

```bash
# Базовая сборка
npm run build

# Сборка с оптимизацией
npm run build:prod

# Сборка с анализом бандла
npm run build:analyze

# Сборка с типизацией
npm run build:types
2. Сборка через TypeScript
bash
# Транспиляция TypeScript
npx tsc

# Сборка с картами исходников
npx tsc --sourceMap

# Сборка в отдельную директорию
npx tsc --outDir dist
3. Сборка через esbuild
bash
# Сборка для Node.js
npx esbuild src/index.ts --bundle --platform=node --outfile=dist/index.js

# Сборка с минификацией
npx esbuild src/index.ts --bundle --platform=node --minify --outfile=dist/index.js

rollup.config.js
javascript
import typescript from '@rollup/plugin-typescript'
import { nodeResolve } from '@rollup/plugin-node-resolve'
import commonjs from '@rollup/plugin-commonjs'
import terser from '@rollup/plugin-terser'

export default [
    // ESM build
    {
        input: 'src/index.ts',
        output: {
            file: 'dist/index.mjs',
            format: 'esm',
            sourcemap: true
        },
        plugins: [
            nodeResolve(),
            commonjs(),
            typescript({
                tsconfig: './tsconfig.module.json'
            })
        ]
    },
    // CJS build
    {
        input: 'src/index.ts',
        output: {
            file: 'dist/index.js',
            format: 'cjs',
            sourcemap: true
        },
        plugins: [
            nodeResolve(),
            commonjs(),
            typescript({
                tsconfig: './tsconfig.build.json'
            })
        ]
    },
    // Minified CJS
    {
        input: 'src/index.ts',
        output: {
            file: 'dist/index.min.js',
            format: 'cjs',
            sourcemap: true
        },
        plugins: [
            nodeResolve(),
            commonjs(),
            typescript({
                tsconfig: './tsconfig.build.json'
            }),
            terser()
        ]
    }
]
🚀 Скрипты сборки
package.json
json
{
    "scripts": {
        "build": "npm run build:clean && npm run build:main",
        "build:clean": "rimraf dist coverage",
        "build:main": "tsc -p tsconfig.build.json",
        "build:esm": "tsc -p tsconfig.module.json",
        "build:cjs": "tsc -p tsconfig.json",
        "build:types": "tsc -p tsconfig.json --emitDeclarationOnly",
        "build:min": "npm run build:cjs && terser dist/index.js -o dist/index.min.js",
        "build:analyze": "npm run build && source-map-explorer dist/index.js",
        "build:watch": "tsc -p tsconfig.build.json --watch"
    }
}
🐳 Docker сборка
Dockerfile
dockerfile
# ============================================
# BUILD STAGE
# ============================================
FROM node:18-alpine AS builder

WORKDIR /app

COPY package*.json ./
RUN npm ci --only=production

COPY . .
RUN npm run build

# ============================================
# PRODUCTION STAGE
# ============================================
FROM node:18-alpine

WORKDIR /app

COPY --from=builder /app/dist ./dist
COPY --from=builder /app/node_modules ./node_modules
COPY --from=builder /app/package*.json ./

ENV NODE_ENV=production
ENV PORT=3000

EXPOSE 3000

CMD ["node", "dist/index.js"]
Сборка Docker образа
bash
# Сборка образа
docker build -t fab-shield:latest .

# Сборка с указанием версии
docker build -t fab-shield:1.0.0 .

# Сборка для production
docker build --target production -t fab-shield:prod .

# Просмотр слоев
docker history fab-shield:latest

🚨 Устранение проблем
Ошибка сборки
bash
# Очистка кэша
npm cache clean --force

# Удаление node_modules
rm -rf node_modules

# Установка заново
npm install

# Повторная сборка
npm run build
Проблемы с TypeScript
bash
# Проверка типов
npx tsc --noEmit

# Проверка конкретного файла
npx tsc src/index.ts --noEmit

# Сброс кэша TypeScript
rm -rf node_modules/.cache
Проблемы с памятью
bash
# Увеличение памяти для Node.js
node --max-old-space-size=4096 node_modules/.bin/tsc

# Использование ts-node с увеличенной памятью
NODE_OPTIONS="--max-old-space-size=4096" npx ts-node build.ts
