## 🧪 Тесты производительности

### 1. Базовый тест

```typescript
// tests/performance/basic.test.ts

import { FABShield } from '@fab-registry/shield'
import express from 'express'
import request from 'supertest'

describe('Performance Tests', () => {
    test('should handle 10000 requests under 1 second', async () => {
        const app = express()
        const shield = new FABShield()
        app.use(shield.middleware())
        app.get('/test', (req, res) => res.json({ ok: true }))
        
        const start = Date.now()
        const requests = Array(10000).fill(null).map(() => 
            request(app).get('/test')
        )
        
        await Promise.all(requests)
        const duration = Date.now() - start
        
        expect(duration).toBeLessThan(1000)
    })
})
2. Тест задержки
typescript
// tests/performance/latency.test.ts

describe('Latency Tests', () => {
    test('should add less than 5ms overhead', async () => {
        const app = express()
        
        // Без FAB Shield
        app.get('/no-shield', (req, res) => res.json({ ok: true }))
        
        // С FAB Shield
        const shield = new FABShield()
        app.use(shield.middleware())
        app.get('/with-shield', (req, res) => res.json({ ok: true }))
        
        const noShieldTime = await measureLatency('/no-shield')
        const withShieldTime = await measureLatency('/with-shield')
        
        const overhead = withShieldTime - noShieldTime
        expect(overhead).toBeLessThan(5)
    })
})
🔧 Оптимизация
1. Кэширование
typescript
// Включение кэширования
const shield = new FABShield({
    cache: {
        enabled: true,
        ttl: 3600,
        maxSize: 1000
    }
})
2. Ленивая загрузка
typescript
// Включение ленивой загрузки
const shield = new FABShield({
    performance: {
        lazyLoading: true,
        preloadModules: ['headers', 'csp']
    }
})
3. Параллельная обработка
typescript
// Включение параллельной обработки
const shield = new FABShield({
    performance: {
        parallel: {
            enabled: true,
            maxConcurrency: 4
        }
    }
})
4. Минимальная конфигурация
typescript
// Минимальная конфигурация для лучшей производительности
const shield = new FABShield({
    headers: { enabled: true },
    csp: { enabled: false },
    ai: { enabled: false },
    monitoring: { enabled: false }
})
